Object

Table with properties and methods determined by its class. Indexes can be read and written to access properties. 

local Car = Class("Car")

--[[
    The method init() is special because it is called immediately after an
    object is created and before the object is returned.
]]
function Car:init(wheels, doors)
    self.Wheels = wheels or 4
    self.Doors = doors or 4
    self.Speed = 0
end

function Car:Accelerate()
    self.Speed += 1
end

function Car:Decelerate()
    self.Speed -= 1
end

function Car:Brake()
    self.Speed = 0
end

local car = Car(4, 2)
print(car.Wheels) -- 4
print(car.Doors) -- 2
print(car.Speed) -- 0
